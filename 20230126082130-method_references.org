#+title: Method References

=Printer.java=
#+begin_src java
public interface Printer {
    void print(String message, int age);
}
#+end_src

* Static member:
=LambdasDemo.java=
#+begin_src java
public class LambdasDemo {
   public static void print(String message, int age){
    System.out.println(message + age);
   }

   public static void show() {
    greet((message, age) -> print(message, age));
    // OR as a Method reference
    greet(LambdasDemo::print);
   }

   public static void greet(Printer printer){
    printer.print("Hello World", 69);
   }
}
#+end_src
