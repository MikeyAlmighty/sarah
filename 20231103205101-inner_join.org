#+title: INNER JOIN

=INNER JOIN= keyword SELECTS records that have matching values in *both* tables.

#+BEGIN_SRC sql
SELECT ProductID, ProductName, CategoryName
FROM Products
INNER JOIN Categories ON Products.CategoryID = Categories.CategoryID;
#+END_SRC
