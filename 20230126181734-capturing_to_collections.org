* Summing to a Predicate

#+begin_src java
public static void show() {
        Predicate<Movie> greaterThan10 = movie -> movie.getLikes() > 10;
        var movieStream = Stream.of(
                new Movie("b", 15),
                new Movie("a", 5),
                new Movie("d", 15),
                new Movie("e", 35),
                new Movie("c", 5)
            );

             var result = movieStream
                .filter(greaterThan10)
                .collect(Collectors.toMap(Movie::getTitle, Function.identity()));
                // .collect(Collectors.toMap(Movie::getTitle, m -> m)); // ^ Same as

            System.out.println(result.toString()); // {b=com.demo.Movie@c387f44, d=com.demo.Movie@4e0e2f2a, e=com.demo.Movie@73d16e93}
}
#+end_src

* Joining by delimiter
#+begin_src java
public static void show() {
    Predicate<Movie> greaterThan10 = movie -> movie.getLikes() > 10;
    var movieStream = Stream.of(
            new Movie("b", 15),
            new Movie("a", 5),
            new Movie("d", 15),
            new Movie("e", 35),
            new Movie("c", 5)
        );

            var result = movieStream
            .filter(greaterThan10)
            .map(Movie::getTitle)
            .collect(Collectors.joining("&"));

        System.out.println(result.toString()); //  b&d&e
}
#+end_src
