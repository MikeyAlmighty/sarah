#+title: Executor Framework

=Thread pool= - consists of (n) worker threads

(Threads aren't (destroyed/created) they're always existant they just become occupied or freed.)

=ThreadPoolExecutor=

#+begin_src java
public static void show () {
    var executor = Executors.newFixedThreadPool(2); // Pool with 2 workers
    try {
        executor.submit(() -> {
            System.out.println(Thread.currentThread().getName());
        });
    } finally {
        executor.shutdown(); // Terminate Stops new tasks
        // executor.shutdownNow(); // Stops current Task and new tasks
    }
}
#+end_src
