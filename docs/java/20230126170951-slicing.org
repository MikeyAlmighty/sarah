 #+begin_src java
public static void show() {
    Predicate<Movie> highestRated = movie -> movie.getLikes() < 26;
    var stream = Stream.of(
            new Movie("Deadpool 1", 15),
            new Movie("Deadpool 2", 5),
            new Movie("Deadpool 3", 25),
            new Movie("Shrek 1", 35),
            new Movie("Shrek 2", 5)
        );
        stream
            .takeWhile(highestRated)
            .forEach(movie -> System.out.println(movie.getTitle())); // Deadpool 1 Deadpool 2 Deadpool 3
}
#+end_src

 #+begin_src java
public static void show() {
    Predicate<Movie> highestRated = movie -> movie.getLikes() < 25;
    var stream = Stream.of(
            new Movie("Deadpool 1", 15),
            new Movie("Deadpool 2", 5),
            new Movie("Deadpool 3", 25),
            new Movie("Shrek 1", 35),
            new Movie("Shrek 2", 5)
        );
        stream
            .dropWhile(highestRated)
            .forEach(movie -> System.out.println(movie.getTitle())); // Deadpool 3 Shrek 1 Shrek 2
}
