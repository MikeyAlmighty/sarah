#+begin_src java
public static void show() {
    Predicate<Movie> isPopular = movie -> movie.getLikes() > 10;
    var stream = Stream.of(
         new Movie("Deadpool 1", 15),
         new Movie("Deadpool 2", 5),
         new Movie("Deadpool 3", 25)
        );
        stream
            .filter(isPopular)
            .forEach(highRated -> System.out.println(highRated.getTitle()));
}
