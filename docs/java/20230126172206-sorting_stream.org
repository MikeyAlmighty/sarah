    public static void show() {
        var movieStream = Stream.of(
                new Movie("b", 15),
                new Movie("a", 5),
                new Movie("d", 25),
                new Movie("e", 35),
                new Movie("c", 5)
            );
            movieStream
                .sorted(Comparator.comparing(Movie::getTitle))
                // .sorted((a, b) -> a.getTitle().compareTo(b.getTitle())) //  ^Same as
                // .sorted(Comparator.comparing(m -> m.getTitle())) // ^Same as
                .forEach(movie -> System.out.println(movie.getTitle()));
    }
// To reverse: .sorted(Comparator.comparing(Movie::getTitle).reversed())
