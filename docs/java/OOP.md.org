public class Pen {
    // Attributes
    String type = "Gel";
    String color = "Blue";
    byte point = 10;
    static boolean clicked = false;

    // Actions
    public static void click() {
        clicked = true;
    }

    public static void unclick() {
        clicked = false;
    }
}

#+begin_src java
// Hero
public class Hero {
  public static void kick() {
      System.out.println("Kick");
  }
  
  public static void punch() {
      System.out.println("Punch");
  }

  public static void powerUp() {
    System.out.println("Power Up!");
  }

  public static void powerDown() {
    System.out.println("Power Down!");
  }
}
#+end_src

#+begin_src java
// Hero1
public class Hero1 extends Hero {
    String suitColor = "Red";

    public static void setPower(int power) {
        System.out.println(power);
    }
}
#+end_src

#+begin_src java
// Hero2
public class Hero2 extends Hero {

    public static void setAgility(int agility) {
        System.out.println(agility);
    }
}
#+end_src

#+begin_src java
// Main
public class Main {
    
    public static void main(String[] args) {
       Hero1 hero1 = new Hero1();
       Hero2 hero2 = new Hero2();
       hero2.setAgility(20);
    //    hero1.setAgility(20); // Not allowed as setAgility does not exist on Hero1
    //    hero2.setPower(20); // Not allowed as setPower does not exist on Hero2
       hero1.setPower(15);
       hero1.punch();
       hero1.kick();
       hero1.powerDown();
       hero1.powerUp();
    }
}
#+end_src
