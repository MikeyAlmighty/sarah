public class ListDemo {
    public static void show(){
        List<String> list = new ArrayList<>();
        list.add("DeadPool");
        list.add("Batman");
        list.add("AquaMan");
        list.add(1, "Robin");
        Collections.addAll(list, "A", "B", "C");
        System.out.println(list);

        System.out.println(list.get(2));

        // Replace
        list.set(2, "Robin Hood");

        // from (incl) to (excl)
        list.subList(0, 2);
    }
}
