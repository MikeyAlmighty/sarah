* Selection Sort
In place comparison sorting algorithm that keeps track of minimum value during each iteration, at the end of each iteration swap variables.

`Time complexity`: O(n<sup>2</sup>)

#+BEGIN_SRC js
const selectionSort = (array) => {
  const arr = array.slice()
  for (let i = 0; i < arr.length - 1; i++) {
    let min = i
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[min] > arr[j]) {
        min = j
      }
    }
    let temp = arr[i]
    arr[i] = arr[min]
    arr[min] = temp
  }
  return arr
}

const array = [2, 19, 4, 25, 6, 520]

console.log(selectionSort(array))
