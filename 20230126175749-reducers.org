public static void show() {
        Predicate<Movie> greaterThan15 = movie -> movie.getLikes() > 20;
        var movieStream = Stream.of(
                new Movie("b", 15),
                new Movie("a", 5),
                new Movie("d", 15),
                new Movie("e", 35),
                new Movie("c", 5)
            );
            var great = movieStream
              .anyMatch(greaterThan15);
            System.out.println(great); // true
}

* General Purpose reduction
#+begin_src java
public static void show() {
  var movieStream = Stream.of(
        new Movie("b", 15),
        new Movie("a", 5),
        new Movie("d", 15),
        new Movie("e", 35),
        new Movie("c", 5)
    );
    Optional<Integer> totalLikes = movieStream
        .map(Movie::getLikes)
        .reduce(Integer::sum);
    System.out.println(totalLikes.orElse(0)); // 75
}
